23static int ft_atoi(const char *nptr)
24{
25 int number;
26 short int minus_sign;

28 number = 0;
29 minus_sign = 0;
30 while (*nptr != '\0' && (*nptr == ' ' || *nptr == '\t' || *nptr == '\n'

32 nptr++;
33 if (*nptr != '\0' && ((*nptr == '-') || (*nptr == '+')))
