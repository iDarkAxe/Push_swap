25void parse_and_execute(t_data *data)
26{
27 char *str;
28
29 str = get_next_line(0);
30 while (str != NULL)
31 {
32 if (ft_strlen(str) < 3 || (*str != 'p' && *str != 's' && *str != 'r'))
33 return ;
34 if (*str == 'p' && *(str + 1) == 'a' && *(str + 2) == '\n')
35 ft_pa(data, PRINT_NAME);
36 else if (*str == 'p' && *(str + 1) == 'b' && *(str + 2) == '\n')
37 ft_pb(data, PRINT_NAME);
38 else if (*str == 's' && *(str + 1) == 'a' && *(str + 2) == '\n')
39 ft_sa(data, PRINT_NAME);
40 else if (*str == 's' && *(str + 1) == 'b' && *(str + 2) == '\n')
41 ft_sb(data, PRINT_NAME);
42 else
43 parse_and_execute_rotate(data, str);
44 free(str);
45 str = get_next_line(0);
46 }
47 return ;
48}
