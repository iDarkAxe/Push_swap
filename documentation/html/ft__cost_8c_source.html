24int find_value(t_stack *stack, size_t len, t_bool max_selected)
25{
26 size_t i;
27 int value;
28 t_stack *temp;
29
30 i = 0;
31 value = stack->value;
32 temp = stack;
33 while (temp != NULL && i < len)
34 {
35 if (max_selected == 1 && temp->value > value)
36 value = temp->value;
37 else if (max_selected == 0 && temp->value < value)
38 value = temp->value;
39 temp = temp->next;
40 i++;
41 }
42 return (value);
43}
