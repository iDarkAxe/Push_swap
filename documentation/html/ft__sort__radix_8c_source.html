46static int count_max_bits(t_stack **stack, size_t len)
47{
48 t_stack *head;
49 size_t count;
50 size_t max;
51 int max_bits;
52
53 head = *stack;
54 max = head->index;
55 max_bits = 0;
56 count = 0;
57 while (head && count < len)
58 {
59 if (head->index > max)
60 max = head->index;
61 head = head->next;
62 count++;
63 }
64 while ((max >> max_bits) != 0)
65 max_bits++;
66 return (max_bits);
