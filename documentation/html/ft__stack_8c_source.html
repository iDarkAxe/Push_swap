72void ft_stackadd_back(t_stack **stack, t_stack *new, size_t stack_len)
73{
74 t_stack *last;
75 t_stack *stack_cpy;
76 size_t index;
77
78 index = 0;
79 if (stack == NULL || new == NULL)
80 return ;
81 last = NULL;
82 stack_cpy = *stack;
83 while (stack_cpy != NULL && index < stack_len - 1)
84 {
85 last = stack_cpy;
86 stack_cpy = stack_cpy->next;
87 index++;
88 }
89 if (last == NULL)
90 *stack = new;
91 else
92 {
93 last->next = new;
94 new->prev = last;
95 new->next = *stack;
96 (*stack)->prev = new;
97 }
98}
