23static size_t find_value_index(t_stack *stack, size_t len)
24{
25 size_t count;
26 t_stack *temp;
27
28 count = 0;
29 temp = stack;
30 while (temp != NULL && len > 0)
31 {
32 if (temp->value < stack->value)
33 count++;
34 temp = temp->next;
35 len--;
36 }
37 return (count);
38}
